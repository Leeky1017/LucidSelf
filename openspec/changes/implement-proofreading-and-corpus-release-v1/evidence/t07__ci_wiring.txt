#!/usr/bin/env bash
set -euo pipefail

usage() {
  cat <<'USAGE'
Gate-2: Proofreading QA + Corpus Release Pipeline

Usage:
  bash scripts/gates/gate2_proofreading_and_corpus_release.sh [--skip-openspec]

Runs:
  - openspec validate --specs --strict --no-interactive
  - scripts/gates/asset_gate_corpus.sh
  - scripts/gates/gate1_cross_book_conflicts_and_quality.sh
  - scripts/proofreading_qa/run_workflow.py run
  - scripts/corpus_release_pipeline/release.py publish + validate + rollback

Artifacts:
  - Exports LS_ARTIFACT_DIR (default: artifacts)
  - Writes:
      artifacts/proofreading_qa/qa_queue.json
      artifacts/proofreading_qa/qa_report.json
      artifacts/proofreading_qa/qa_report.md
      artifacts/corpus_releases/<corpus_release_id>/release_manifest.json
      artifacts/corpus_releases/<corpus_release_id>/hashes.json
      artifacts/corpus_releases/active_release.json
USAGE
}

SKIP_OPENSPEC=0

for arg in "$@"; do
  case "$arg" in
    -h|--help)
      usage
      exit 0
      ;;
    --skip-openspec)
      SKIP_OPENSPEC=1
      ;;
    *)
      echo "Unknown argument: $arg" >&2
      echo >&2
      usage >&2
      exit 2
      ;;
  esac
done

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/../.." && pwd)"
cd "$ROOT_DIR"

PYTHON_BIN=""
if [[ -x ".venv/bin/python" ]]; then
  PYTHON_BIN=".venv/bin/python"
elif command -v python3 >/dev/null 2>&1; then
  PYTHON_BIN="python3"
elif command -v python >/dev/null 2>&1; then
  PYTHON_BIN="python"
else
  echo "Missing python interpreter (expected .venv/bin/python, python3, or python)" >&2
  exit 127
fi

if [[ -z "${LS_ARTIFACT_DIR:-}" ]]; then
  export LS_ARTIFACT_DIR="artifacts"
fi
mkdir -p "$LS_ARTIFACT_DIR/proofreading_qa" "$LS_ARTIFACT_DIR/corpus_releases"

if [[ -z "${LS_VERSION_ID:-}" ]] && command -v git >/dev/null 2>&1; then
  SHA="$(git rev-parse --short=12 HEAD 2>/dev/null || true)"
  if [[ -n "$SHA" ]]; then
    export LS_VERSION_ID="ver_${SHA}"
  fi
fi

if [[ -z "${LS_CORPUS_RELEASE_ID:-}" ]] && command -v git >/dev/null 2>&1; then
  TREE="$(git rev-parse HEAD:data 2>/dev/null || true)"
  if [[ -n "$TREE" ]]; then
    export LS_CORPUS_RELEASE_ID="cr_${TREE:0:12}"
  fi
fi

if [[ -z "${LS_VERSION_ID:-}" ]]; then
  export LS_VERSION_ID="$(PYTHONPATH=. $PYTHON_BIN -c 'from backend.core.versioning.ids import resolve_version_id; print(resolve_version_id())')"
fi

if [[ -z "${LS_CORPUS_RELEASE_ID:-}" ]]; then
  export LS_CORPUS_RELEASE_ID="$(PYTHONPATH=. $PYTHON_BIN -c 'from backend.core.versioning.ids import resolve_corpus_release_id; print(resolve_corpus_release_id())')"
fi

if [[ "$SKIP_OPENSPEC" -eq 0 ]]; then
  if ! command -v openspec >/dev/null 2>&1; then
    echo "Missing 'openspec' CLI in PATH" >&2
    exit 127
  fi
  echo "\$ openspec validate --specs --strict --no-interactive"
  openspec validate --specs --strict --no-interactive
fi

echo "\$ bash scripts/gates/asset_gate_corpus.sh --skip-openspec"
bash scripts/gates/asset_gate_corpus.sh --skip-openspec

echo "\$ bash scripts/gates/gate1_cross_book_conflicts_and_quality.sh --skip-openspec"
bash scripts/gates/gate1_cross_book_conflicts_and_quality.sh --skip-openspec

echo "\$ ${PYTHON_BIN} scripts/proofreading_qa/run_workflow.py run --policy data/proofreading_qa/qa_policy.v1.yaml --output-dir ${LS_ARTIFACT_DIR}/proofreading_qa"
PYTHONPATH=. $PYTHON_BIN scripts/proofreading_qa/run_workflow.py run \
  --policy data/proofreading_qa/qa_policy.v1.yaml \
  --output-dir "${LS_ARTIFACT_DIR}/proofreading_qa"

echo "\$ ${PYTHON_BIN} scripts/corpus_release_pipeline/release.py publish --artifact-root ${LS_ARTIFACT_DIR} --release-root ${LS_ARTIFACT_DIR}/corpus_releases --force"
PYTHONPATH=. $PYTHON_BIN scripts/corpus_release_pipeline/release.py publish \
  --artifact-root "${LS_ARTIFACT_DIR}" \
  --release-root "${LS_ARTIFACT_DIR}/corpus_releases" \
  --force

echo "\$ ${PYTHON_BIN} scripts/corpus_release_pipeline/release.py validate --release-root ${LS_ARTIFACT_DIR}/corpus_releases --corpus-release-id ${LS_CORPUS_RELEASE_ID}"
PYTHONPATH=. $PYTHON_BIN scripts/corpus_release_pipeline/release.py validate \
  --release-root "${LS_ARTIFACT_DIR}/corpus_releases" \
  --corpus-release-id "${LS_CORPUS_RELEASE_ID}"

echo "\$ ${PYTHON_BIN} scripts/corpus_release_pipeline/release.py rollback --release-root ${LS_ARTIFACT_DIR}/corpus_releases --to ${LS_CORPUS_RELEASE_ID} --reason gate2_smoke_rollback"
PYTHONPATH=. $PYTHON_BIN scripts/corpus_release_pipeline/release.py rollback \
  --release-root "${LS_ARTIFACT_DIR}/corpus_releases" \
  --to "${LS_CORPUS_RELEASE_ID}" \
  --reason gate2_smoke_rollback

# Gate-2 workflow: proofreading QA + corpus release pipeline

name: Gate-2 Proofreading QA + Corpus Release Pipeline

on:
  push:
    branches: [main, develop]
    paths:
      - 'data/**'
      - 'openspec/**'
      - 'scripts/**'
      - '.github/workflows/gate2-proofreading-and-corpus-release.yml'
  pull_request:
    branches: [main, develop]
    paths:
      - 'data/**'
      - 'openspec/**'
      - 'scripts/**'
      - '.github/workflows/gate2-proofreading-and-corpus-release.yml'
  workflow_dispatch: {}

env:
  PYTHON_VERSION: '3.12'

jobs:
  gate2:
    name: Gate-2 (QA + Release)
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Node
        uses: actions/setup-node@v4
        with:
          node-version: '22'

      - name: Install OpenSpec CLI
        run: |
          npm install -g @fission-ai/openspec@0.17.2
          openspec --version

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: 'pip'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run Gate-2 acceptance
        run: |
          bash scripts/gates/gate2_proofreading_and_corpus_release.sh

      - name: Upload Gate-2 artifacts
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: gate2-proofreading-and-corpus-release-artifacts
          path: artifacts/
          retention-days: 30
